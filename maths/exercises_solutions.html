def standardize(X):
    """Centre each feature to mean=0 and scale to std=1."""
    mean = X.mean(axis=0)
    std  = X.std(axis=0)
    return (X - mean) / std

X_std = standardize(X)
print("Mean of standardised X (should be ~0):",  np.round(X_std.mean(axis=0), 6))
print("Std  of standardised X (should be ~1):",  np.round(X_std.std(axis=0),  6))

pca = PCA(n_components=2)
Z_sklearn = pca.fit_transform(X_std)

for i in range(2):
    match_direct = np.allclose(Z_manual[:, i],  Z_sklearn[:, i], atol=1e-6)
    match_flipped = np.allclose(Z_manual[:, i], -Z_sklearn[:, i], atol=1e-6)
    status = " match" if (match_direct or match_flipped) else " mismatch"
    print(f"PC{i+1}: {status}  (direct={match_direct}, sign-flipped={match_flipped})")

print("\nCorrelation between manual and sklearn PCs:")
for i in range(2):
    corr = np.abs(np.corrcoef(Z_manual[:, i], Z_sklearn[:, i])[0, 1])
    print(f"  PC{i+1}: |corr| = {corr:.6f}  (should be 1.0)")
